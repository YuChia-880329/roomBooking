SELECT
	ID,
  	CLIENT_ID,
  	ROOM_ID,
  	ROOM_NUM,
  	PAY_METHOD_ID,
  	CHECKIN_DATETIME,
  	CHECKOUT_DATE,
  	USE_DAY,
  	TOTAL_MONEY
FROM(

	SELECT 
  		O.ID,
  		O.CLIENT_ID,
  		O.ROOM_ID,
  		O.ROOM_NUM,
  		O.PAY_METHOD_ID,
  		O.CHECKIN_DATETIME,
  		O.CHECKOUT_DATE,
  		O.USE_DAY,
  		O.TOTAL_MONEY,
  		ROW_NUMBER()OVER(ORDER BY O.ID DESC) RN
	FROM(
	
  		SELECT 
  			O.ID,
  			O.CLIENT_ID,
  			O.ROOM_ID,
  			O.ROOM_NUM,
  			O.PAY_METHOD_ID,
  			O.CHECKIN_DATETIME,
  			O.CHECKOUT_DATE,
  			O.USE_DAY,
  			O.TOTAL_MONEY,
  			R.HOTEL_ID
		FROM 
  			BOOKING_ORDER O
		JOIN 
  			ROOM R
		ON
  			O.ROOM_ID = R.ID
		WHERE
  			R.HOTEL_ID = 1
  			
  		) O
  		
  	) 
WHERE RN BETWEEN 1 AND 1
